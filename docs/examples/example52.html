Example 5.2 - Mouse CLick

This is a live example. The code is embedded in the web page. View the page source with your browser to see it all. 

    //////////////////////////
    // Lesson 5.2 Example
    // Comments start with "//"
    
    // Let's revisit the Blink game one more time.
    // We will use game states to change the color.
    // The catch is, we will change states with a mouse
    // click instead of in the update() function.

    // These are the possible game states:
    var mode = {
        RED: 0,
        GREEN : 1,
        BLUE : 2,
    };
    
    // This is the variable used to keep track of the current state:
    var game_state = mode.RED;

    // We drawn on the HTML5 canvas.
    var canvas = document.getElementById("canvas");
    
    // The "context" variable has functions for drawing on the screen.
    var context = canvas.getContext("2d");
    
    // We will draw some text later.
    context.font="14px monospace";
    context.textAlign = "center";

    // game_core() is our main game function.
    function game_core() {
       update();  // check the rules
       draw();    // draw the game on the screen
       vfx();     // display visual effects
       sfx();     // play sound effects
    } // game_core()

    // update() checks the game rules.
    function update() {
        // No rules - they were moved below to the onclick function.
    } // update()
    
    // The canvas is an element on the web page.
    // As such, it knowns about the mouse.
    // We want to tell the canvas what to do when it
    // sees a mouse click.
    // Since a function is a list of commands to execute at
    // a later time, we will bundle up our instructions into
    // a function.
    // The tricky part here is that the function does not need a name.
    // We never call the function, so we don't care about the
    // function's name.
    // This type of function is called an unnamed,
    // anonymous, or lambda function.
    canvas.onclick = function() {
        // The "switch" statement checks a variable and then jumps
        // to a "case" statement matching the value.
        // The "break" statement will exit the switch statement.
        // We use "break" at the end of each "case" block.
        switch (game_state) {
            case mode.RED:
                game_state = mode.GREEN;  // RED changes to GREEN
                break;
            case mode.GREEN:
                game_state = mode.BLUE;   // GREEN changes to BLUE
                break;
            case mode.BLUE:
                game_state = mode.RED;    // BLUE changes to RED
                break;
        } // switch
    } // onclick()
    
    // draw() draws on the screen.
    function draw() {

        // We are going to draw a background color that matches
        // the current game state.
        switch (game_state) {
            case mode.RED:
                context.fillStyle = "#f00";        // #f00 is red.
                context.fillRect(0, 0, 200, 150);  // Draw a rectangle.
                break;
            case mode.GREEN:
                context.fillStyle = "#0f0";        // #0f0 is green.
                context.fillRect(0, 0, 200, 150);  // Draw a rectangle.
                break;
            case mode.BLUE:
                context.fillStyle = "#00f";        // #000 is blue.
                context.fillRect(0, 0, 200, 150);  // Draw a rectangle.
                break;
        } // switch

        context.fillStyle = "#000";        // #000 is black.
        context.fillText("You can make games!", 100, 75);

    } // draw()
    
    function vfx() {
        // We have no visual special effects.
    } // vfx()
    
    function sfx() {
        // We have no sound effects.
    } // sfx()
    
    // Call the game_core function every 16ms (i.e., every 0.016 seconds).
    // 16ms per draw is roughly 60 frames per second.
    setInterval(game_core, 16);
